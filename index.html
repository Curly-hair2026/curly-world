    function submitTip(event) {
      event.preventDefault();
      var name = document.getElementById('name').value;
      var tipText = document.getElementById('userTip').value;

      var newTip = { name: name, text: tipText };

      // جلب النصائح السابقة وتخزين الجديدة معها
      var savedTips = JSON.parse(localStorage.getItem('curlyUserTips')) || [];
      savedTips.unshift(newTip);
      localStorage.setItem('curlyUserTips', JSON.stringify(savedTips));

      renderTip(newTip, true);

      document.getElementById('name').value = '';
      document.getElementById('userTip').value = '';

    function renderTip(tip, isNew = false) {
