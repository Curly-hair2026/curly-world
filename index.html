    <script>
        // إعدادات قاعدة البيانات السحابية (JSONBin)
        const BIN_ID = '67389a9be41b4d34e4028fa6';
        const API_KEY = '$2a$10$vNJSJv1y9l1S9Z.M82Wd/eTh7M06N4C6Uj2aZ0Y5NfTqE1/W28gM6';
        const API_URL = `https://api.jsonbin.io/v3/b/${BIN_ID}`;

        // تغيير المظهر (داكن / فاتح)
        function toggleTheme() {
            const currentTheme = document.documentElement.getAttribute('data-theme');
            const newTheme = currentTheme === 'dark' ? 'light' : 'dark';
            document.documentElement.setAttribute('data-theme', newTheme);
        }

        // جلب النصائح من السيرفر السحابي
        async function fetchTips() {
            try {
                const response = await fetch(API_URL + '/latest', {
                    headers: { 'X-Access-Key': API_KEY }
                });
                const data = await response.json();
                const tips = data.record.tips || [];
                renderTips(tips);
            } catch (error) {
                console.error("خطأ في جلب النصائح:", error);
                document.getElementById('tipsContainer').innerHTML = '<p style="color: red;">تعذر تحميل النصائح، تأكدي من الاتصال بالإنترنت.</p>';
            }
        }

        // عرض النصائح في الصفحة
        function renderTips(tips) {
            const container = document.getElementById('tipsContainer');
            if (tips.length === 0) {
                container.innerHTML = '<p style="color: #888;">لا توجد نصائح بعد، كوني أول من يشارك!</p>';
                return;
            }
            container.innerHTML = '';
            tips.slice().reverse().forEach(tipText => {
                const tipDiv = document.createElement('div');
                tipDiv.className = 'tip-item';
                tipDiv.textContent = tipText;
                container.appendChild(tipDiv);
            });
        }

        // إضافة نصيحة جديدة وإرسالها للسيرفر
        async function addTip() {
            const input = document.getElementById('tipInput');
            const newTip = input.value.trim();

            if (!newTip) {
                alert('الرجاء كتابة نصيحة أولاً!');
                return;
            }

            try {
                // جلب النصائح القديمة أولاً
                const getResponse = await fetch(API_URL + '/latest', {
                    headers: { 'X-Access-Key': API_KEY }
                });
                const getData = await getResponse.json();
                let currentTips = getData.record.tips || [];

                // إضافة النصيحة الجديدة
                currentTips.push(newTip);

                // تحديث السيرفر
                await fetch(API_URL, {
                    method: 'PUT',
                    headers: {
                        'Content-Type': 'application/json',
                        'X-Access-Key': API_KEY
                    },
                    body: JSON.stringify({ tips: currentTips })
                });

                input.value = '';
                renderTips(currentTips);
                alert('تمت إضافة نصيحتكِ وتحديث الصفحة عند جميع الزوار! 🎉');

            } catch (error) {
                console.error("خطأ في حفظ النصيحة:", error);
                alert('حدث خطأ أثناء الحفظ، يرجى المحاولة مرة أخرى.');
            }
        }

        // جلب النصائح فور فتح الصفحة
        window.onload = fetchTips;
